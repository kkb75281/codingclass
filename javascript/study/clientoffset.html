09. 요소 메서드 : 크기 및 위치 : element.getBoundungClientRect() : 전체 크기 및 위치(객체로 표현) 

document.querySelector("#sample01 .btn9").addEventListener("click", (e)=>{
    e.preventDefault();

    // document.querySelector("#sample01 .sResult").innerHTML = "이 박스의 getBoundingClientRect()값은 " +box1.getBoundingClientRect() + "px(문서 기준) 입니다.";
    // console.log(box1.getBoundingClientRect());  DOMRect 객체 방식으로 결과 뜸

    const box = document.querySelector("#sample01 .sbox");
    const result = document.querySelector("#sample01 .sResult");
    const rect = box.getBoundingClientRect();

    result.innerHTML = "";

    for(let key in rect){
        if(typeof rect[key] !== "function"){                // 객체 안에 함수가 아닌 것 찾기
            let span = document.createElement("span");      // 데이터 값을 span에 넣어줌
            span.textContent= `${key}: ${rect[key]} `;
            result.appendChild(span);
        };
    };
});
